# Definindo os valores do gráfico
N <- 2:20 #vetor N
texec<- 1:19 #tempo de execução
cpi<- 1#CPI
Inst<- 40 #chute de instruções
T <- 1 / 50 #período = 1/freq

for(i in 0:18){ #calcula o tempo de execução para cada iteração
  InstF = N[i+1]*Inst
  texec[i] = cpi * InstF * T
}

# Graph cars using blue points overlayed by a line 
plot(texec, N, type="o", col="blue") #plota o gráfico

# Create a title with a red, bold/italic font
title(main="texecXn", col.main="black", font.main=4)
